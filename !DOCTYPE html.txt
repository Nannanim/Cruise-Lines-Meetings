<script>
  let timers = {};
  let isRunning = false;

  function formatTime(sec) {
    const m = Math.floor(sec / 60).toString().padStart(2, '0');
    const s = (sec % 60).toString().padStart(2, '0');
    return `${m}:${s}`;
  }

  function startCycle() {
    resetAll();
    startTimer(1);
  }

  function startTimer(id) {
    if (isRunning) return;
    isRunning = true;

    clearInterval(timers[1]);
    clearInterval(timers[2]);

    document.getElementById("timer1").classList.remove("active");
    document.getElementById("timer2").classList.remove("active");
    document.getElementById(`timer${id}`).classList.add("active");

    const min = parseInt(document.getElementById(`min${id}`).value) || 0;
    const sec = parseInt(document.getElementById(`sec${id}`).value) || 0;
    let total = min * 60 + sec;

    const display = document.getElementById(`display${id}`);
    display.style.color = "white";

    timers[id] = setInterval(() => {

      display.textContent = formatTime(total);

      if (total <= 0) {
        clearInterval(timers[id]);
        isRunning = false;
        document.getElementById("alarmSound").play();
        display.textContent = "00:00";
        display.style.color = varYellow;

        const nextId = id === 1 ? 2 : 1;

        if (nextId === 1) {
          setTimeout(() => {
            document.getElementById("cycleSound").play();
          }, 500);
        }

        setTimeout(() => startTimer(nextId), 1000);
      }

      total--;
    }, 1000);
  }

  function resetAll() {
    clearInterval(timers[1]);
    clearInterval(timers[2]);
    document.getElementById("display1").textContent = "00:00";
    document.getElementById("display2").textContent = "00:00";
    document.getElementById("timer1").classList.remove("active");
    document.getElementById("timer2").classList.remove("active");
    isRunning = false;
  }
</script>
